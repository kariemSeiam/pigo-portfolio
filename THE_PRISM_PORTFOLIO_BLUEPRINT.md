**Material Design ripple on click:**

```tsx
// components/adaptive/AdaptiveCursor.tsx (Android variant)
'use client';

import { useEffect, useState, useRef } from 'react';
import { motion, AnimatePresence } from 'framer-motion';

interface Ripple {
  id: string;
  x: number;
  y: number;
}

export function AndroidCursor() {
  const [cursorPosition, setCursorPosition] = useState({ x: 0, y: 0 });
  const [ripples, setRipples] = useState<Ripple[]>([]);
  const rafRef = useRef<number>();

  useEffect(() => {
    // Smooth cursor tracking with RAF
    const handleMouseMove = (e: MouseEvent) => {
      if (rafRef.current) {
        cancelAnimationFrame(rafRef.current);
      }

      rafRef.current = requestAnimationFrame(() => {
        setCursorPosition({ x: e.clientX, y: e.clientY });
      });
    };

    // Create ripple on click
    const handleClick = (e: MouseEvent) => {
      const newRipple: Ripple = {
        id: `ripple-${Date.now()}`,
        x: e.clientX,
        y: e.clientY,
      };

      setRipples((prev) => [...prev, newRipple]);

      // Remove ripple after animation
      setTimeout(() => {
        setRipples((prev) => prev.filter((r) => r.id !== newRipple.id));
```

```tsx
'use client';

import { useEffect, useRef } from 'react';

export function AndroidHexagonPattern() {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    // Set canvas size
    const resize = () => {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
      drawHexagons();
    };

    const drawHexagon = (x: number, y: number, radius: number) => {
      ctx.beginPath();
      for (let i = 0; i < 6; i++) {
        const angle = (Math.PI / 3) * i;
        const hx = x + radius * Math.cos(angle);
        const hy = y + radius * Math.sin(angle);
        if (i === 0) {
          ctx.moveTo(hx, hy);
        } else {
          ctx.lineTo(hx, hy);
        }
      }
      ctx.closePath();
      ctx.strokeStyle = 'rgba(61, 220, 132, 0.1)'; // Android green, very subtle
      ctx.lineWidth = 1;
      ctx.stroke();
    };

    const drawHexagons = () => {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      const hexRadius = 50;
      const hexHeight = hexRadius * Math.sqrt(3);
      const hexWidth = hexRadius * 2;

      const cols = Math.ceil(canvas.width / (hexWidth * 0.75)) + 1;
      const rows = Math.ceil(canvas.height / hexHeight) + 1;

      for (let row = 0; row < rows; row++) {
        for (let col = 0; col < cols; col++) {
          const x = col * hexWidth * 0.75;
          const y = row * hexHeight + (col % 2) * (hexHeight / 2);
          drawHexagon(x, y, hexRadius);
        }
      }
    };

    resize();
    window.addEventListener('resize', resize);

    return () => {
      window.removeEventListener('resize', resize);
    };
  }, []);

  return (
    <canvas
      ref={canvasRef}
      className="fixed inset-0 -z-10 pointer-events-none"
      style={{ opacity: 0.5 }}
```

```tsx
'use client';

import { Canvas, useFrame, useLoader } from '@react-three/fiber';
import { GLTFLoader } from 'three/examples/jsm/loaders/GLTFLoader';
import { useScroll, useTransform } from 'framer-motion';
import { useRef, useEffect } from 'react';
import * as THREE from 'three';

function AndroidModel() {
  const gltf = useLoader(GLTFLoader, '/models/android-robot.glb');
  const meshRef = useRef<THREE.Group>(null);
  const { scrollYProgress } = useScroll();

  // Map scroll progress (0-1) to assembly (parts coming together)
  const assemblyProgress = useTransform(scrollYProgress, [0.2, 0.8], [0, 1]);

  useEffect(() => {
    const unsubscribe = assemblyProgress.on('change', (latest) => {
      if (!meshRef.current) return;

      // Animate parts based on scroll
      meshRef.current.children.forEach((child, index) => {
        const delay = index * 0.1;
        const progress = Math.max(0, Math.min(1, (latest - delay) / (1 - delay)));

        // Move parts from scattered to assembled position
        const startY = 5 - index * 0.5; // Parts start above
        const endY = 0;
        child.position.y = startY + (endY - startY) * progress;

        // Fade in
        if (child instanceof THREE.Mesh) {
          child.material.opacity = progress;
        }

        // Rotate during assembly
        child.rotation.y = (1 - progress) * Math.PI * 2;
      });
    });

    return unsubscribe;
  }, [assemblyProgress]);

  return <primitive ref={meshRef} object={gltf.scene} scale={2} />;
}
```

```tsx
'use client';

import { useEffect, useRef } from 'react';

export function WebGridPattern() {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    let animationId: number;
    let offset = 0;

    const resize = () => {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
    };

    const draw = () => {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      const gridSize = 50;
      const lineWidth = 1;

      // Vertical lines
      for (let x = -offset; x < canvas.width; x += gridSize) {
        ctx.strokeStyle = 'rgba(0, 217, 255, 0.1)';
        ctx.lineWidth = lineWidth;
        ctx.beginPath();
        ctx.moveTo(x, 0);
        ctx.lineTo(x, canvas.height);
        ctx.stroke();
      }

      // Horizontal lines
      for (let y = -offset; y < canvas.height; y += gridSize) {
        ctx.strokeStyle = 'rgba(0, 217, 255, 0.1)';
        ctx.lineWidth = lineWidth;
        ctx.beginPath();
        ctx.moveTo(0, y);
        ctx.lineTo(canvas.width, y);
        ctx.stroke();
      }

      // Animate (slow drift)
      offset = (offset + 0.5) % gridSize;

      animationId = requestAnimationFrame(draw);
    };

    resize();
    draw();
    window.addEventListener('resize', resize);

    return () => {
      window.removeEventListener('resize', resize);
      cancelAnimationFrame(animationId);
    };
  }, []);

  return (
    <canvas
      ref={canvasRef}
      className="fixed inset-0 -z-10 pointer-events-none"
      style={{ opacity: 0.6 }}
    />
  );
```
